    <script>
        lucide.createIcons();

        const progressBar = document.getElementById('scrollProgress');
        const mapLineFill = document.querySelector('.map-line-fill');
        const steps = document.querySelectorAll('.step-trigger');
        const navText = document.getElementById('currentStepText');
        const navbar = document.getElementById('navbar');
        const mapSection = document.getElementById('map-start');

        const observerOptions = {
            threshold: 0.5,
            rootMargin: "-50px"
        };

        const stepObserver = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                const node = entry.target.querySelector('.step-node');
                const stepNum = entry.target.dataset.step;
                
                if (entry.isIntersecting) {
                    node.classList.add('active');
                    if (navText) {
                        navText.innerText = `Explorando Paso ${stepNum}`;
                        navText.style.color = '#E64580';
                        navText.style.fontWeight = 'bold';
                    }
                } else {
                    node.classList.remove('active');
                }
            });
        }, observerOptions);

        steps.forEach(step => stepObserver.observe(step));

        window.addEventListener('scroll', () => {
            const winScroll = document.body.scrollTop || document.documentElement.scrollTop;
            const height = document.documentElement.scrollHeight - document.documentElement.clientHeight;
            const scrolled = (winScroll / height) * 100;
            if (progressBar) progressBar.style.width = scrolled + "%";

            if(winScroll > 50) {
                navbar.classList.add('shadow-md');
            } else {
                navbar.classList.remove('shadow-md');
            }

            if (mapSection && mapLineFill) {
                const mapHeight = mapSection.offsetHeight;
                const mapTop = mapSection.offsetTop;
                const middle = winScroll + (window.innerHeight / 2);

                let fillPercentage = ((middle - mapTop) / mapHeight) * 120;
                if (fillPercentage < 0) fillPercentage = 0;
                if (fillPercentage > 100) fillPercentage = 100;
                
                mapLineFill.style.height = `${fillPercentage}%`;
            }
        });

        // Smooth scroll
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                const target = document.querySelector(this.getAttribute('href'));
                if (!target) return;
                e.preventDefault();
                target.scrollIntoView({ behavior: 'smooth' });
            });
        });
        
        // FAQ
        document.addEventListener("DOMContentLoaded", () => {
            const faqItems = document.querySelectorAll(".faq-item");

            faqItems.forEach(item => {
                const question = item.querySelector(".faq-question");

                question.addEventListener("click", () => {
                    const isActive = item.classList.contains("active");
                    faqItems.forEach(i => i.classList.remove("active"));
                    if (!isActive) item.classList.add("active");
                });
            });
        });
    </script>
